# isObject

Check whether the given value is an object

### Import

```typescript copy
import { isObject } from '@fullstacksjs/toolbox';
```

### Signature

```typescript copy
function isObject(x: unknown): x is Record<number | string | symbol, any> {}
```

### Examples

```typescript copy
 * isObject('')            // false
 * isObject('hello world') // false
 * isObject(null)          // false
 * isObject(true)          // false
 * isObject(undefined)     // false
 * isObject(NaN)           // false
 * isObject(0)             // false
 * isObject(isObject)      // false
 * isObject(false)         // false
 * isObject([])            // false
 * isObject([2])           // false
 * isObject({})            // true
 * isObject({ a: 2 })      // true
 * isObject({ 2: 'a' })    // true
```
